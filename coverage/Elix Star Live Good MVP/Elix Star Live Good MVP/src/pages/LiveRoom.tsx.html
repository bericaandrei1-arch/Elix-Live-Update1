// Full Live Room with Chat, Likes, and Battle Support
// NO AUTHENTICATION REQUIRED - GUEST SUPPORT ENABLED
// AI BOT SYSTEM ENABLED - 1000+ BOTS

import { useEffect, useState, useRef } from 'react';
import { useParams } from 'react-router-dom';
import { supabase } from '../lib/supabase';
import { realtimeService } from '../lib/realtimeService';
import { aiBotSystem } from '../lib/aiBotSystem';
import { Eye, X } from 'lucide-react';

export default function LiveRoom() {
  const { roomId } = useParams();
  const [isHost, setIsHost] = useState(false);
  const [streamData, setStreamData] = useState<any>(null);
  const [viewerCount, setViewerCount] = useState(0);

  // Chat & Likes
  const [messages, setMessages] = useState<any[]>([]);
  const [likeCount, setLikeCount] = useState(0);
  const [inputValue, setInputValue] = useState('');

  // Battle Mode
  const [battleMatch, setBattleMatch] = useState<any>(null);
  const [battleScore, setBattleScore] = useState({ scoreA: 0, scoreB: 0 });

  // Flying Hearts Animation
  const [hearts, setHearts] = useState<{ id: number; x: number; y: number }[]>([]);
  const heartIdRef = useRef(0);
  
  useEffect(() => {
    console.log('[LiveRoom] Mounted, roomId:', roomId);

    if (!roomId) {
      console.error('[LiveRoom] No roomId provided');
      return;
    }

    // NO USER CHECK - GUESTS ALLOWED
    // Subscribe to room events
    const channel = realtimeService.subscribeToRoom(roomId, {
      onChatMessage: (msg) => {
        console.log('[LiveRoom] New chat message:', msg);
        setMessages(prev => [...prev, msg]);
      },
      onLikeUpdate: (like) => {
        console.log('[LiveRoom] Like update:', like);
        setLikeCount(like.total);
        
        // Trigger flying heart animation
        spawnHeart();
      },
    });

    // Start AI Bots (simulate 50-200 viewers)
    const botViewerCount = Math.floor(Math.random() * 150) + 50;
    setViewerCount(botViewerCount);
    aiBotSystem.startBots(roomId, botViewerCount);

    // Random bot likes every 2-5 seconds
    const botLikeInterval = setInterval(() => {
      if (Math.random() < 0.3) { // 30% chance every interval
        aiBotSystem.sendLike(roomId);
      }
    }, Math.random() * 3000 + 2000);

    return () => {
      console.log('[LiveRoom] Unsubscribing from room');
      channel.unsubscribe();
      aiBotSystem.stopBots(roomId);
      clearInterval(botLikeInterval);
    };
  }, [roomId]);
  
  // Subscribe to battle updates if battle is active
  useEffect(() => {
    if (!battleMatch?.id) return;
    
    console.log('[LiveRoom] Subscribing to battle:', battleMatch.id);
    
    const battleChannel = supabase
      .channel(`battle:${battleMatch.id}`)
      .on('postgres_changes', {
        event: 'UPDATE',
        schema: 'public',
        table: 'battle_scores',
        filter: `match_id=eq.${battleMatch.id}`,
      }, (payload) => {
        console.log('[LiveRoom] Battle score update:', payload.new);
        setBattleScore({ scoreA: payload.new.score_a, scoreB: payload.new.score_b });
      })
      .subscribe();
      
    return () => {
      console.log('[LiveRoom] Unsubscribing from battle');
      battleChannel.unsubscribe();
    };
  }, [battleMatch]);
  
  const handleGoLive = () => {
    console.log('[LiveRoom] Go Live clicked');
    // TODO: Start streaming
  };
  
  const handleEndStream = () => {
    console.log('[LiveRoom] End Stream clicked');
    // TODO: End streaming
  };
  
  const handleSendMessage = async () => {
    if (!isHost && inputValue.trim()) {
      console.log('[LiveRoom] Sending message:', inputValue);
      try {
        await realtimeService.sendMessage(roomId!, inputValue);
        setInputValue('');
      } catch (error) {
        console.error('[LiveRoom] Failed to send message:', error);
      }
    }
  };
  
  const handleSendLike = async () => {
    if (!isHost) {
      console.log('[LiveRoom] Sending like');
      try {
        await realtimeService.sendLike(roomId!);
        // Flying heart animation triggered in onLikeUpdate
      } catch (error) {
        console.error('[LiveRoom] Failed to send like:', error);
      }
    }
  };

  // Spawn Flying Heart Animation
  const spawnHeart = () => {
    const id = heartIdRef.current++;
    const x = Math.random() * 80 + 10; // Random X position (10-90%)
    const y = 90; // Start from bottom
    
    setHearts(prev => [...prev, { id, x, y }]);
    
    // Remove heart after animation (3 seconds)
    setTimeout(() => {
      setHearts(prev => prev.filter(h => h.id !== id));
    }, 3000);
  };

  return (
    <div className="h-screen bg-black relative overflow-hidden">
      <div id="video-container" className="w-full h-full" />

      {/* Chat overlay - scrolls upward (TikTok-style) */}
      <div className="absolute left-4 bottom-24 max-h-96 w-64 overflow-y-auto flex flex-col-reverse">
        {messages.map((msg, index) => (
          <div key={msg.id || index} className="bg-black/50 text-white px-3 py-1 rounded mb-2">
            <span className="font-bold text-[#E6B36A]">{msg.username || 'Guest'}: </span>
            <span className="text-white/80">{msg.message_text}</span>
          </div>
        ))}
      </div>

      {/* Like counter + Viewer count */}
      <div className="absolute top-4 left-4 flex flex-col gap-2">
        <div className="flex items-center gap-2 bg-black/50 px-3 py-1 rounded-full">
          <span className="bg-red-500 px-2 py-1 rounded font-bold text-white text-xs">LIVE</span>
          <Eye size={16} className="text-white" />
          <span className="text-white text-sm font-bold">{viewerCount}</span>
        </div>
        <div className="flex items-center gap-2 bg-black/50 px-3 py-1 rounded-full">
          <span className="text-xl">❤️</span>
          <span className="text-white text-sm font-bold">{likeCount}</span>
        </div>
      </div>

      {/* Flying Hearts Animation */}
      {hearts.map(heart => (
        <div
          key={heart.id}
          className="absolute text-4xl animate-float-up pointer-events-none"
          style={{
            left: `${heart.x}%`,
            bottom: `${heart.y}%`,
            animation: 'floatUp 3s ease-out forwards',
          }}
        >
          ❤️
        </div>
      ))}

      {/* Battle Score (when battle active) */}
      {battleMatch && (
        <div className="absolute top-20 left-1/2 transform -translate-x-1/2 bg-black/70 px-6 py-3 rounded-full text-white">
          <span className="text-2xl font-bold text-blue-400">{battleScore.scoreA}</span>
          <span className="mx-4 text-white">VS</span>
          <span className="text-2xl font-bold text-red-400">{battleScore.scoreB}</span>
        </div>
      )}

      {/* Exit Button (Top Right) */}
      <button
        onClick={() => window.history.back()}
        className="absolute top-4 right-4 w-10 h-10 bg-black/50 rounded-full flex items-center justify-center text-white hover:bg-black/70 transition"
      >
        <X size={24} />
      </button>

      {/* Viewer controls (chat input + like button) */}
      {!isHost && (
        <div className="absolute bottom-4 left-4 right-4 flex gap-2">
          <input
            value={inputValue}
            onChange={(e) => setInputValue(e.target.value)}
            placeholder="Say something..."
            className="flex-1 px-4 py-2 rounded-full bg-white/20 text-white placeholder-white/50 border-none outline-none"
            onKeyDown={(e) => e.key === 'Enter' && handleSendMessage()}
          />
          <button
            onClick={handleSendMessage}
            className="px-6 py-2 bg-blue-500 rounded-full font-bold text-white hover:bg-blue-600 transition"
          >
            Send
          </button>
          <button
            onClick={handleSendLike}
            className="px-4 py-2 bg-red-500 rounded-full text-xl hover:bg-red-600 transition"
          >
            ❤️
          </button>
        </div>
      )}

      {/* Host controls (no chat input, per TikTok rules) */}
      {isHost && (
        <div className="absolute bottom-4 right-4 flex gap-2">
          <button
            onClick={handleGoLive}
            className="bg-red-500 text-white px-6 py-3 rounded-full font-bold hover:bg-red-600 transition"
          >
            Go Live
          </button>
          <button
            onClick={handleEndStream}
            className="bg-gray-700 text-white px-6 py-3 rounded-full font-bold hover:bg-gray-800 transition"
          >
            End Stream
          </button>
        </div>
      )}

      {/* CSS Animation for Flying Hearts */}
      <style>{`
        @keyframes floatUp {
          0% {
            transform: translateY(0) scale(1);
            opacity: 1;
          }
          50% {
            transform: translateY(-200px) scale(1.2);
            opacity: 0.8;
          }
          100% {
            transform: translateY(-400px) scale(0.5);
            opacity: 0;
          }
        }
      `}</style>
    </div>
  );
}
